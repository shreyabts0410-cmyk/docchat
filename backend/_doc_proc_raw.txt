    try:
        from reportlab.lib.pagesizes import letter

        return b, False

    out = io.BytesIO()
    doc = SimpleDocTemplate(out, pagesize=letter)
    styles = getSampleStyleSheet()
    flow = []
    for line in new_text.split("\n"):
